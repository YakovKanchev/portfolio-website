21     public Response createItem(ItemDTO itemDTO){
22         return  RestAssured.given()
23                 .log().all() //Log everything (url, headers, body)
24                 .contentType(ContentType.JSON) //Sets content-type: application/json
25                 .accept(ContentType.JSON) //Sets accept: application/json
26                 .auth().oauth2(token) //Sets bearer token in the request header
27                 .headers("User-Agent", "Mozilla Firefox")

29                 .basePath(BASE_PATH) // Set base path
30                 .when()
31                 .body(gson.toJson(itemDTO))
